#!/usr/bin/env bash

rm -rf ./venv
if [ "$1" == "python3" ];then
    python3 ./virtualenv.py -p python3  --no-pip  --no-wheel --no-setuptools venv
else
    python ./virtualenv.py -p python  --no-pip  --no-wheel --no-setuptools venv
fi

source venv/bin/activate
cd ./install/setuptools
./setup.py install
cd ../meld 
python ./setup.py install
cd ../supervisor 
python ./setup.py install
